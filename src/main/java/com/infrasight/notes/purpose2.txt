import React, { useEffect, useState } from 'react';
import axiosInstance from '../api/axiosInstance';
import { UserPoints } from '../types/UserPoints';

import {
  Card,
  CardContent,
  Typography,
  Container,
  Box,
  Stack,
} from '@mui/material';

function UserPointDisplay() {
  const [point, setPoint] = useState<UserPoints | null>(null);

  useEffect(() => {
    axiosInstance.get<UserPoints>('/api/point')
      .then((res) => setPoint(res.data))
      .catch((error) => console.error('API Error:', error));
  }, []);

  return (
    <Container maxWidth="sm" sx={{ mt: 4 }}>
      <Typography variant="h4" gutterBottom>
        User Point Info
      </Typography>

      {point ? (
        <Stack direction="row" spacing={2}>
          {[
            { label: 'Name', value: point.name },
            { label: 'X', value: point.x },
            { label: 'Y', value: point.y },
            { label: 'Z', value: point.z },
          ].map((item) => (
            <Card key={item.label} sx={{ minWidth: 120, textAlign: 'center', boxShadow: 3 }}>
              <CardContent>
                <Typography variant="subtitle1" fontWeight="bold">
                  {item.label}
                </Typography>
                <Typography variant="h6">{item.value}</Typography>
              </CardContent>
            </Card>
          ))}
        </Stack>
      ) : (
        <Typography variant="body1">Loading...</Typography>
      )}
    </Container>
  );
}

export default UserPointDisplay;
